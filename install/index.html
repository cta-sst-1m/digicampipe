Continous integration with Travis¶

Travis CI is a popular continious integration system.

Bellow the .travis.yml file that is used to run Travis. This is executed at each pull request.

language: generic


matrix:
  include:
    - os: linux
      python: 3.5
      env:
        - PYTHON_VERSION=3.5
        - MINICONDA_URL=https://repo.continuum.io/miniconda/Miniconda3-latest-Linux-x86_64.sh
    - os: linux
      python: 3.6
      env:
        - PYTHON_VERSION=3.6
        - MINICONDA_URL=https://repo.continuum.io/miniconda/Miniconda3-latest-Linux-x86_64.sh
    - os: osx
      python: 3.5
      env:
        - PYTHON_VERSION=3.5
        - MINICONDA_URL=https://repo.continuum.io/miniconda/Miniconda3-latest-MacOSX-x86_64.sh
    - os: osx
      python: 3.6
      env:
        - PYTHON_VERSION=3.6
        - MINICONDA_URL=https://repo.continuum.io/miniconda/Miniconda3-latest-MacOSX-x86_64.sh


before_install:
  - export MPLBACKEND='Agg'

install:
- wget $MINICONDA_URL -O miniconda.sh
- bash miniconda.sh -b -p $HOME/miniconda
- export PATH="$HOME/miniconda/bin:$PATH"
- hash -r
- conda config --set always_yes yes --set changeps1 no
- conda update -q conda
- conda info -a

- conda create -q -n digicampipe python=$PYTHON_VERSION pip
- conda env update -f environment.yml
- source activate digicampipe
- pip install --upgrade pip
- pip install .
script:
- pytest digicampipe
- make -C docs/ html
- |
  if [ "${TRAVIS_EVENT_TYPE}" == "cron" ]; then
    travis_wait 30 pytest -c all_tests.ini digicampipe
  fi
